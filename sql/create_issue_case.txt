create table issue_case (
        case_id varchar(255),
        type_id int,
        commit_hash_new varchar(255),
        committer_new varchar(255),
        time_new datetime,
        commit_hash_last varchar(255),
        file_path_last varchar(255),
        commit_hash_disappear varchar(255), 
        committer_disappear varchar(255),
        time_disappear datetime,
        create_time datetime,
        update_time datetime,
        case_status varchar(255),
        primary key(case_id),
        foreign key(type_id) references issue_type(type_id)
        )
