create table issue_location (
        location_id varchar(255),
        instance_id varchar(255),
        file_path varchar(255),
        class varchar(255),
        method varchar(255),
        start_line int,
        end_line int,
        start_offset int,
        end_offset int,
        code varchar(255),
        msg varchar(255),
        primary key(location_id),
        foreign key(instance_id) references issue_instance(instance_id)
        )
