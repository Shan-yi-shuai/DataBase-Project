create table issue_location (
        location_id int AUTO_INCREMENT,
        instance_id int,
        file_path varchar(255),
        class varchar(255),
        method varchar(255),
        start_line int,
        end_line int,
        start_offset int,
        end_offset int,
        code varchar(255),
        msg varchar(255),
        records varchar(255),
        include_records varchar(1023),
        primary key(location_id),
        foreign key(instance_id) references issue_instance(instance_id)
        )
